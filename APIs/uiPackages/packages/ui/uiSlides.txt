$uiSlides = ($slideIndexRef ? Mutable ; $_slides ? (List:Any)) fun {
	$currentSlideRef ? Var = 0 at _slides.
	
	slideIndexRef = 0.
	slideIndexRef listen {currentSlideRef = x at _slides}.
	
	uiShowVariable: currentSlideRef
}.
$uiLazySlides = ($slideIndexRef ? Mutable ; $_slideGenerator) fun {
	$currentSlideRef ? Var = _slideGenerator: 0.
	
	slideIndexRef = 0.
	slideIndexRef listen {currentSlideRef = _slideGenerator: x}.
	
	uiShowVariable: currentSlideRef
}.
