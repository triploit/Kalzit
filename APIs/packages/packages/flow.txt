$switch = ($value ? Any ; $_array ? (List:Any)) fun {{1 at x at _array} each {(0 at x at _array): value} filter arrayIndexes: _array}.
$ifElse = ($condition ; $_ie ? (2 SizedList Any)) fun {
	do: ((0 eq do: condition) at _ie)
}.
$categorize = ($_items ? (List:Any) ; $_categories ? (List:2 SizedList Any)) fun {($_x fun {_x switch categories}) each _items}.
$doEach = ($f ; $_argSets ? (List:List:Any)) fun {($_x fun {f do _x}) each _argSets}.


$lastFlowCondition = true. $lastFlowConditionRef = reference: $lastFlowCondition.

$if = ($condition ; $then) fun {$result = condition ifElse then;{void}. lastFlowConditionRef = condition. result}.
$ifNot = ($condition ; $then) fun {$result = condition ifElse {void};then. lastFlowConditionRef = condition. result}.
$else = ($then) fun {!ifNot lastFlowCondition then}.
